# Configuring the Installation Disk as a Repo

# This procedure is NOT required on the exam. 

# Use the dd command to create a raw image of the installation media
example: dd if=/dev/sr* of=/rhel.iso bs=1M

# Create a mount point for the ISO
example: mkdir /repo

# Back up /etc/fstab as a safety measure
example: cp /etc/fstab /etc/fstab.bak

# Add the ISO to /etc/fstab for automatic mounting
# Use echo with shell redirection to append the mount entry
example: echo "/rhel.iso /repo iso9660 defaults 0 0" >> /etc/fstab

# Mount all filesystems defined in /etc/fstab
example: mount -a
